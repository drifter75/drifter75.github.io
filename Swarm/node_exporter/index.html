    node-exporter, CADvisor & postgres-exporter in Docker Swarm

    Source: Okami101 Blog

    script to export node names to prometheus
    node_exporter_entrypoint
    #!/bin/sh -e

NODE_NAME=$(cat /etc/nodename)
echo "node_meta{node_id=\"$NODE_ID\", container_label_com_docker_swarm_node_id=\"$NODE_ID\", node_name=\"$NODE_NAME\"} 1" > /home/node-meta.prom

set -- /bin/node_exporter "$@"

exec "$@"
    deployment yml for node-exporter & CAdvisor
    version: '3.8'
#
networks:
  metrics-network:
    driver: overlay
  traefik-network:
    external: true
configs:
  node_exporter_entrypoint:
    file: ./configs/node-exporter/node_exporter_entrypoint
#
services:
  cadvisor:
    image: gcr.io/cadvisor/cadvisor:v0.47.2
    hostname: "{{.Node.Hostname}}"
    environment:
      - --housekeeping_interval=30s
      - --max_housekeeping_interval=35s
      - --global_housekeeping_interval=30s
      - --docker_only=true
      - --store_container_labels=false
      # - --disable_metrics=percpu,process,sched,tcp,udp,diskIO,disk,network
      - --disable_metrics=accelerator,cpu_topology,disk,memory_numa,tcp,udp,percpu,sched,process,hugetlb,referenced_memory,resctrl,cpuset,advtcp,memory_numa
      - --storage_duration=60s
#
      - --event_storage_event_limit=default=0
      - --event_storage_age_limit=default=0
    # ports:
    # - published: 8080
      # target: 8080
      # mode: host
    volumes:
      - /:/rootfs:ro
      - /var/run/docker.sock:/var/run/docker.sock:ro
      - /var/run:/var/run:rw
      - /sys:/sys:ro
      - /var/lib/docker:/var/lib/docker:ro
      - /sys/fs/cgroup:/cgroup:ro
      - /dev/disk/:/dev/disk:ro
      - /dev/kmsg:/dev/kmsg
    networks:
      - metrics-network
      - traefik-network
    deploy:
      mode: global
      # resources:
        # limits:
          # cpus: '0.35'
          # memory: 100M
        # reservations:
          # cpus: '0.15'
          # memory: 128M
######
  node-exporter:
    image: quay.io/prometheus/node-exporter:v1.7.0
    environment:
      - NODE_ID={{.Node.ID}}
    command: 
      - '--path.procfs=/host/proc'
      - '--path.rootfs=/rootfs'
      - '--path.sysfs=/host/sys'
      - '--collector.filesystem.mount-points-exclude=^/(sys|proc|dev|host|etc)($$|/)'
      - '--collector.textfile.directory=/home'
    volumes:
      - /:/host:ro,rslave
      - /proc:/host/proc:ro
      - /sys:/host/sys:ro
      - /:/rootfs:ro
      - /etc/hostname:/etc/nodename
    configs:
      - source: node_exporter_entrypoint
        target: /docker-entrypoint.sh
    entrypoint:
      - /bin/sh
      - /docker-entrypoint.sh
    networks:
      - metrics-network
      - traefik-network
    deploy:
      mode: global

##
  postgres-exporter:
    image: quay.io/prometheuscommunity/postgres-exporter
    secrets:
      - postgres15-password
    environment:
      TZ: Europe/Berlin
      DATA_SOURCE_URI: postgres15:5432/postgres?sslmode=disable
      DATA_SOURCE_USER: postgres15
#      DATA_SOURCE_PASS: ${PG_DB_PASSWORD}
      DATA_SOURCE_PASS_FILE:  /run/secrets/postgres15-password
    networks:
      - metrics-network
      - traefik-network
    volumes:
      - /etc/hosts:/etc/hosts
    ports:
      - "9187:9187"
    deploy:
      mode: replicated
      replicas: 1
      placement:
        max_replicas_per_node: 1
        constraints:
          - node.hostname == db_server01

secrets:
  postgres15-password:
    external: true
    Add node-exporter and CADvisor jobs to prometheus.yml
      - job_name: "cadvisor"
    dns_sd_configs:
      - names:
          - "tasks.cadvisor"
        type: "A"
        port: 8080

  - job_name: "node-exporter"
    dns_sd_configs:
      - names:
          - "tasks.node-exporter"
        type: "A"
        port: 9100

  - job_name: "postgres-exporter-data-01"
    static_configs:
      - targets: ["postgres-exporter:9187"]
