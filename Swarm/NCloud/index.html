    Create DB & User
    CREATE DATABASE nextclouddb;
CREATE USER ncdbuser WITH PASSWORD 'My$ecret7ass';
GRANT ALL PRIVILEGES ON DATABASE nextclouddb TO ncdbuser;
ALTER DATABASE nextclouddb OWNER TO ncdbuser;
GRANT USAGE ON SCHEMA public TO ncdbuser;
    NC admin password
    printf "My$ecret7ass" | docker secret create nextcloud-admin-password -

    Create Volume for next cloud data
    docker volume create --driver local --opt type=none --opt device=/home/cargo/nc_nextcloud-data --opt o=bind nextcloud-data
    deploy nextcloud-swarm.yml
    nextcloud-swarm.yml
    version: '3.8'

x-default-opts:
  &default-opts
  # logging:
    # options:
      # max-size: "10m"
  logging:
    driver: "none"

networks:
  traefik-network:
    external: true

volumes:
  nextcloud-data:
    external: true

secrets:
  nextcloud-postgres-password:
    external: true
  nextcloud-admin-password:
    external: true

services:
  nextcloud:
    <<: *default-opts
    image: nextcloud
    volumes:
      - nextcloud-data:/var/www/html
    environment:
      TZ: "Europe/Berlin"
      POSTGRES_HOST: postgres15
      DB_PORT: 5432
      POSTGRES_DB: nextclouddb
      POSTGRES_USER: nextclouddbuser
      POSTGRES_PASSWORD_FILE: /run/secrets/nextcloud-postgres-password
#
      NEXTCLOUD_ADMIN_USER: ncadmin
      NEXTCLOUD_ADMIN_PASSWORD_FILE: /run/secrets/nextcloud-admin-password
      NEXTCLOUD_TRUSTED_DOMAINS: next.example.com
      TRUSTED_PROXIES: next.example.com
      OVERWRITECLIURL: https://next.example.com
      OVERWRITEPROTOCOL: https
      OVERWRITEHOST: next.example.com
      REDIS_HOST: redis
      PHP_MEMORY_LIMIT: 1024M
      PHP_UPLOAD_LIMIT: 1024M
      MEMORY_LIMIT: 1024M
      UPLOAD_MAX_SIZE: 1024M
#
    networks:
      - traefik-network
    secrets:
      - nextcloud-postgres-password
      - nextcloud-admin-password
    deploy:
      mode: replicated
      replicas: 1
      placement:
        max_replicas_per_node: 1
        constraints:
          - node.hostname == Server001
      resources:
        limits:
          cpus: '1.55'
          memory: 2G
        reservations:
          cpus: '0.55'
          memory: 512M
      labels:
        - "traefik.enable=true"
        - "traefik.http.routers.nextcloud.rule=Host(`next.example.com`)"
        - "traefik.http.routers.nextcloud.service=nextcloud"
        - "traefik.http.routers.nextcloud.entrypoints=websecure"
        - "traefik.http.services.nextcloud.loadbalancer.server.port=80"
        - "traefik.http.routers.nextcloud.tls=true"
        - "traefik.http.routers.nextcloud.tls.certresolver=letsencrypt"
        - "traefik.http.services.nextcloud.loadbalancer.passhostheader=true"
        - "traefik.http.routers.nextcloud.middlewares=compresstraefik"
        - "traefik.http.middlewares.compresstraefik.compress=true"
    Nextcloud CRON
    DOCKID=$(docker ps -aqf "name=nc_nextcloud") && docker exec -t -u www-data $DOCKID php -f /var/www/html/cron.php >/dev/null 2>&1
