    bash script

    splnk
    #!/bin/bash
###########
counter=$1
i=1
host=$(hostname)

while [ $i -le $counter ]
do
    THEDATE=`date +"%Y%m%d.%H:%M:%S"`
    message="$(echo -n $THEDATE) Test log for Splunk from $(echo -n $host) , this is $i of $counter "
    curl -s http://192.168.1.85:8088/services/collector/event -H "Authorization: Splunk 1b627160-517c-443d-ba98-fce35c00f2df" -d '{ "event":"'"$message"'"}' > /dev/null
    i=$(( i + 1 ))
done
    splnk_delayed
    #!/bin/bash
###########
counter=$1
i=1
host=$(hostname)

while [ $i -le $counter ]
do
    THEDATE=`date +"%Y%m%d.%H:%M:%S"`
    message="$(echo -n $THEDATE) Test log for Splunk from $(echo -n $host) , this is $i of $counter "
    curl -s http://192.168.1.85:8088/services/collector/event -H "Authorization: Splunk 1b627160-517c-443d-ba98-fce35c00f2df" -d '{ "event":"'"$message"'"}' > /dev/null
    i=$(( i + 1 )) ; sleep 1
done
